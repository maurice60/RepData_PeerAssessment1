zip <- 'activity.zip'
activity <- read.csv(unz(zip, 'activity.csv'))
daily <- aggregate(activity['steps'], by=activity['date'], FUN='sum', na.rm=TRUE)
bySlice <- aggregate(activity['steps'], by=activity['interval'], FUN='mean', na.rm=TRUE)
